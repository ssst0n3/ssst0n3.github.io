import tqdm.notebook as tqdm
from tqdm import trange

bit = 512
offset = 16

def stage1_binary_search(N, gift):
    def check(q, k):
        qh = q >> offset
        p = (gift & (2^k-1)) ^^ qh
        return  (p * q - N) % 2^k == 0

    def binary_search(ql):
        q = ql
        for k in range(bit-offset):
            # q_{k+1} = q_k
            if not check(q, k+1):
                # q_{k+1} = q_k + 2^k
                q = q + 2^(k+offset)
                if not check(q, k+1):
                    return False
        if N % q == 0:
            print("found")
            return q
    
    def solve():
        box = []
        ql = 2
        while ql.nbits() < 16:
            box.append(ql)
            ql = ql.next_prime()
        box += list(set(range(3,2^16,2)) - set(box))
        for ql in tqdm.tqdm(box):
            q = binary_search(ql)
            if q:
                return q

    q = solve()
    assert N % q == 0
    p = N // q
    return p
