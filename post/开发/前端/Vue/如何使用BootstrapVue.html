如何使用BootstrapVue

一、使用WebStorm创建Vue项目

二、在main.js引用BootstrapVue

import BootstrapVue from 'bootstrap-vue'
import 'bootstrap/dist/css/bootstrap.css'
import 'bootstrap-vue/dist/bootstrap-vue.css'
Vue.use(BootstrapVue)
